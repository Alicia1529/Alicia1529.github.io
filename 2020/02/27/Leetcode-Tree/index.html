108. Convert Sorted Array to Binary Search Tree

class Solution {
public TreeNode sortedArrayToBST(int[] nums) {
if (nums.length == 0) return null;
int rootIdx = nums.length/2;
TreeNode root = new TreeNode(nums[rootIdx]);
constructLeftTree(root, nums, 0, rootIdx);
constructRightTree(root, nums, rootIdx + 1, nums.length);
return root;

}
public void constructLeftTree(TreeNode node, int[] nums, int low, int high) {
    if (low >= high) return;
    int nodeIdx = (low + high)/2;
    TreeNode left = new TreeNode(nums[nodeIdx]);
    node.left = left;
    constructLeftTree(left, nums, low, nodeIdx);
    constructRightTree(left, nums, nodeIdx + 1, high);       
}
public void constructRightTree(TreeNode node, int[] nums, int low, int high) {
    if (low >= high) return;
    int nodeIdx = (low + high)/2;
    TreeNode right = new TreeNode(nums[nodeIdx]);
    node.right = right; 
    constructLeftTree(right, nums, low, nodeIdx);
    constructRightTree(right, nums, nodeIdx + 1, high);      
}

}
